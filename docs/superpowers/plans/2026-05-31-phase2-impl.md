```js
async function loadGLB(gender) {
    const key = gender === 'F' ? 'female' : 'male';
    if (_glbCache.has(key)) return _glbCache.get(key);
    const loader = new GLTFLoader();
    const gltf = await loader.loadAsync(`/mannequin_editor/assets/${key}.glb`);
    // Build name → node map
    const nodeMap = new Map();
    gltf.scene.traverse(obj => { if (obj.name) nodeMap.set(obj.name, obj); });
    _glbCache.set(key, nodeMap);
    return nodeMap;
}

function makeToonMaterial(color) {
    return new THREE.MeshToonMaterial({ color });
}

/**
 * Build segment groups for all bones from GLB meshes.
 * Returns Map<boneName, THREE.Group>
 */
export async function buildSegments(gender) {
    const key = gender === 'F' ? 'female' : 'male';
    const map = MESH_MAP[key];
    const nodeMap = await loadGLB(gender);
    const groups = new Map();

    for (const boneName of BONE_NAMES) {
        const group = new THREE.Group();
        group.name = boneName;

        // Joint sphere — always present, clickable selection handle
        const sphere = new THREE.Mesh(
            new THREE.SphereGeometry(0.04, 12, 8),
            makeToonMaterial(JOINT_COLOR)
        );
        sphere.userData.boneName = boneName;
        sphere.userData.isJoint = true;
        group.add(sphere);

        // GLB mesh segment — clone and re-material
        const glbNodeName = map[boneName];
        if (glbNodeName) {
            const glbNode = nodeMap.get(glbNodeName);
            if (glbNode && glbNode.isMesh) {
                const seg = glbNode.clone();
                seg.material = makeToonMaterial(SEGMENT_COLOR);
                seg.userData.boneName = boneName;
                // Reset local transform — position/rotation comes from our bone hierarchy
                seg.position.set(0, 0, 0);
                seg.rotation.set(0, 0, 0);
                seg.scale.setScalar(WORLD_HEIGHT);
                group.add(seg);
            }
        }

        groups.set(boneName, group);
    }
    return groups;
}

/**
 * Compute local offsets for each bone from GLB world positions.
 * Falls back to mannequin-model PROPORTIONS-based offsets if GLB node missing.
 * Returns Map<boneName, THREE.Vector3>
 */
export async function computeBoneOffsets(gender) {
    const key = gender === 'F' ? 'female' : 'male';
    const map = MESH_MAP[key];
    const nodeMap = await loadGLB(gender);

    // Helper: get world position of a GLB node
    function worldPos(glbNodeName) {
        const node = glbNodeName ? nodeMap.get(glbNodeName) : null;
        if (!node) return null;
        const v = new THREE.Vector3();
        node.getWorldPosition(v);
        return v.multiplyScalar(WORLD_HEIGHT);
    }

    // Build offsets from GLB world positions scaled to WORLD_HEIGHT
    // torso is the root, positioned at world origin in our scene
    const offsets = new Map();

    // Compute torso Y from foot world position
    const footPos = worldPos(map.foot_L);
    const torsoPos = worldPos(map.spine) ?? new THREE.Vector3(0, WORLD_HEIGHT * 0.45, 0);
    const torsoY = footPos ? -footPos.y + torsoPos.y : WORLD_HEIGHT * 0.45;

    offsets.set('torso',  new THREE.Vector3(0, torsoY, 0));

    // All other bones: world position relative to their parent
    // We use the GLB world positions directly as absolute positions in our scene
    // then let the hierarchy handle relative offsets.
    // Since we rebuild hierarchy from scratch, store world positions
    // and compute relative in renderer._buildHierarchy.
    // SIMPLIFICATION: store as world offset from scene root (0,0,0)
    // Renderer will re-parent, so store as child-local via subtraction.

    for (const boneName of BONE_NAMES) {
        if (boneName === 'torso') continue;
        const pos = worldPos(map[boneName]);
        if (pos) {
            offsets.set(boneName, pos.clone().add(new THREE.Vector3(0, torsoY, 0)));
        }
    }

    // Fill any missing bones with zero offset (they'll inherit parent position)
    for (const boneName of BONE_NAMES) {
        if (!offsets.has(boneName)) offsets.set(boneName, new THREE.Vector3(0, 0, 0));
    }

    return offsets;
}

export { SELECT_COLOR, JOINT_COLOR };
```

**Note on scale:** GLB models are exported in Blender units (typically 1 unit = 1 meter, mannequin ~1.7m tall). We scale segments by `WORLD_HEIGHT` (2.0) so the mannequin occupies 2 world units. The GLB world positions need the same scale factor.

```js
function sobelCanny(sourceCanvas) {
    const W = sourceCanvas.width;
    const H = sourceCanvas.height;
    const tmp = document.createElement('canvas');
    tmp.width = W; tmp.height = H;
    const ctx = tmp.getContext('2d');
    ctx.drawImage(sourceCanvas, 0, 0);
    const src = ctx.getImageData(0, 0, W, H).data;
    const out = new Uint8ClampedArray(W * H * 4);

    // Sobel kernels
    const kx = [-1, 0, 1, -2, 0, 2, -1, 0, 1];
    const ky = [-1, -2, -1, 0, 0, 0, 1, 2, 1];

    for (let y = 1; y < H - 1; y++) {
        for (let x = 1; x < W - 1; x++) {
            let gx = 0, gy = 0;
            for (let dy = -1; dy <= 1; dy++) {
                for (let dx = -1; dx <= 1; dx++) {
                    const i = ((y + dy) * W + (x + dx)) * 4;
                    const lum = 0.299 * src[i] + 0.587 * src[i + 1] + 0.114 * src[i + 2];
                    const ki = (dy + 1) * 3 + (dx + 1);
                    gx += lum * kx[ki];
                    gy += lum * ky[ki];
                }
            }
            const mag = Math.sqrt(gx * gx + gy * gy);
            const v = mag > 30 ? 255 : 0;
            const di = (y * W + x) * 4;
            out[di] = out[di + 1] = out[di + 2] = v;
            out[di + 3] = 255;
        }
    }

    const imgData = ctx.createImageData(W, H);
    imgData.data.set(out);
    ctx.putImageData(imgData, 0, 0);
    return tmp.toDataURL('image/png');
}
```

Create `static/src/pose-library.js`:

```js
const STORAGE_KEY = 'mannequin_poses';

export class PoseLibrary {
    constructor(editor, renderer) {
        this._editor   = editor;
        this._renderer = renderer;
        this._panel    = null;
        this._visible  = false;
    }

    _load() {
        try { return JSON.parse(localStorage.getItem(STORAGE_KEY) ?? '[]'); }
        catch { return []; }
    }

    _save(poses) {
        localStorage.setItem(STORAGE_KEY, JSON.stringify(poses));
    }

    saveCurrent(name) {
        const scene     = this._editor.getSceneData();
        const thumbnail = this._renderer._renderer.domElement.toDataURL('image/jpeg', 0.6);
        const poses     = this._load();
        poses.unshift({ id: `${Date.now()}`, name, thumbnail, scene });
        // Cap at 50 saved poses to avoid localStorage bloat
        this._save(poses.slice(0, 50));
        this._renderList();
    }

    deletePose(id) {
        this._save(this._load().filter(p => p.id !== id));
        this._renderList();
    }

    loadPose(id) {
        const pose = this._load().find(p => p.id === id);
        if (!pose) return;
        this._editor._renderer.applyScene(pose.scene);
        this._editor._renderer.markDirty();
    }

    toggle() {
        this._visible = !this._visible;
        if (this._panel) this._panel.style.display = this._visible ? '' : 'none';
    }

    mount(container) {
        this._panel = document.createElement('div');
        this._panel.style.cssText = [
            'position:fixed', 'right:0', 'top:0', 'bottom:0', 'width:220px',
            'background:#222', 'border-left:1px solid #444', 'overflow-y:auto',
            'z-index:100', 'display:none', 'flex-direction:column',
        ].join(';');

        const header = document.createElement('div');
        header.style.cssText = 'padding:8px;border-bottom:1px solid #444;flex-shrink:0;';
        header.innerHTML = '<span style="color:#fff;font-size:12px;font-weight:bold">Poses</span>';

        const saveBtn = document.createElement('button');
        saveBtn.textContent = '+ Save current';
        saveBtn.style.cssText = 'width:100%;margin-top:6px;padding:6px;background:#1565c0;color:#fff;border:none;border-radius:4px;cursor:pointer;font-size:11px;';
        saveBtn.onclick = () => {
            const name = prompt('Pose name:', 'Pose');
            if (name?.trim()) this.saveCurrent(name.trim());
        };
        header.appendChild(saveBtn);

        this._listEl = document.createElement('div');
        this._listEl.style.cssText = 'flex:1;overflow-y:auto;padding:4px;';

        this._panel.appendChild(header);
        this._panel.appendChild(this._listEl);
        container.appendChild(this._panel);
        this._renderList();
    }

    _renderList() {
        if (!this._listEl) return;
        const poses = this._load();
        this._listEl.innerHTML = '';
        if (poses.length === 0) {
            this._listEl.innerHTML = '<p style="color:#666;font-size:11px;padding:8px;">No saved poses.</p>';
            return;
        }
        for (const pose of poses) {
            const item = document.createElement('div');
            item.style.cssText = 'display:flex;align-items:center;gap:6px;padding:4px;border-bottom:1px solid #333;cursor:pointer;';
            item.onmouseenter = () => { item.style.background = '#333'; };
            item.onmouseleave = () => { item.style.background = ''; };

            const thumb = document.createElement('img');
            thumb.src = pose.thumbnail;
            thumb.style.cssText = 'width:48px;height:64px;object-fit:cover;border-radius:3px;flex-shrink:0;';
            thumb.onclick = () => this.loadPose(pose.id);

            const info = document.createElement('div');
            info.style.cssText = 'flex:1;min-width:0;';
            info.innerHTML = `<div style="color:#eee;font-size:11px;overflow:hidden;text-overflow:ellipsis;white-space:nowrap">${pose.name}</div>`;

            const del = document.createElement('button');
            del.textContent = '×';
            del.style.cssText = 'background:none;border:none;color:#888;cursor:pointer;font-size:16px;padding:0 4px;flex-shrink:0;';
            del.onclick = (e) => { e.stopPropagation(); this.deletePose(pose.id); };

            item.appendChild(thumb);
            item.appendChild(info);
            item.appendChild(del);
            item.onclick = () => this.loadPose(pose.id);
            this._listEl.appendChild(item);
        }
    }
}
```

```js
mirrorPose(direction = 'L_to_R') {
    const pairs = [
        ['shoulder_L', 'shoulder_R'],
        ['upper_arm_L', 'upper_arm_R'],
        ['forearm_L',   'forearm_R'],
        ['hand_L',      'hand_R'],
        ['thigh_L',     'thigh_R'],
        ['shin_L',      'shin_R'],
        ['foot_L',      'foot_R'],
    ];
    for (const [l, r] of pairs) {
        const [src, dst] = direction === 'L_to_R' ? [l, r] : [r, l];
        const srcBone = this._renderer.bones.get(src);
        const dstBone = this._renderer.bones.get(dst);
        if (!srcBone || !dstBone) continue;
        const q = srcBone.quaternion;
        dstBone.quaternion.set(-q.x, q.y, q.z, q.w);
    }
    this._saveUndoSnapshot();
    this._renderer.markDirty();
}
```
